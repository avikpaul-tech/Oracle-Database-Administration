Chk_Vip_Conn()
{
Source_DB;
hname=`hostname`
LSNR_NAME=`ps -ef|grep tns|grep $DBNAME|egrep -iv '_mvdc|_dg|_asm|_em|_dr'|awk -F 'tnslsnr' '{print $2}'|awk '{print $1}'`
HOST=`cat /etc/hosts|grep $hname|grep vip|awk '{print $NF}'`
PROT=`srvctl config listener -l $LSNR_NAME|grep TCP|awk -F': ' '{print $2}'|awk -F':' '{print $1}'`
PORT=`srvctl config listener -l $LSNR_NAME|grep TCP|awk -F': ' '{print $2}'|awk -F':' '{print $2}'`
connect_string="(DESCRIPTION=(ADDRESS=(PROTOCOL=$PROT)(HOST=$HOST)(PORT=$PORT))(CONNECT_DATA=(SERVICE_NAME=$SRV_NAME)(INSTANCE_NAME=$ORA_SID)))"
username=dbsnmp
password=`/auto/hosting/bin/Oracle/pwVault/toolPwd $DBNAME,$username`
if [[ -z "$password" ]];then
    echo "Result-dbsnmp Password shows null for $DBNAME from password vault|2"
    exit;
fi
Vip_Chk=$(sqlplus -S "$username/$password@$connect_string" <<EOF
set echo off verify off head off feed off pages0 trimspool on;
SELECT 'YES' FROM dual;
EXIT;
EOF
)
export Vip_Chk
}
